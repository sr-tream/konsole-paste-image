cmake_minimum_required(VERSION 3.20)
project(konsole-paste-image VERSION 0.1 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_POSITION_INDEPENDENT_CODE ON)

find_package(ECM 6.0.0 REQUIRED NO_MODULE)
set(CMAKE_MODULE_PATH ${ECM_MODULE_PATH})

include(KDEInstallDirs)
include(KDECMakeSettings)
include(KDECompilerSettings NO_POLICY_SCOPE)

find_package(Qt6 6.5 REQUIRED COMPONENTS Core Gui Widgets)
find_package(KF6 6.0 REQUIRED COMPONENTS CoreAddons I18n XmlGui)

if(NOT KONSOLE_SOURCE_DIR OR NOT EXISTS "${KONSOLE_SOURCE_DIR}/src/pluginsystem/IKonsolePlugin.h")
    message(FATAL_ERROR
        "Set -DKONSOLE_SOURCE_DIR=/path/to/konsole-source (a checkout matching the installed konsole).")
endif()

# Konsole installs libkonsoleapp.so / libkonsoleprivate.so but not the matching
# generated export headers. Stub them: we only need the macros to expand to a
# visible attribute so external linkage works.
file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/konsoleapp_export.h"
"#pragma once
#define KONSOLEAPP_EXPORT __attribute__((visibility(\"default\")))
#define KONSOLEAPP_NO_EXPORT __attribute__((visibility(\"hidden\")))
#define KONSOLEAPP_DEPRECATED __attribute__((deprecated))
#define KONSOLEAPP_DEPRECATED_EXPORT KONSOLEAPP_EXPORT KONSOLEAPP_DEPRECATED
#define KONSOLEAPP_DEPRECATED_NO_EXPORT KONSOLEAPP_NO_EXPORT KONSOLEAPP_DEPRECATED
")

# config-konsole.h is generated by Konsole's own build. We don't actually use
# any of these macros, but headers we include reference them, so stub all 0.
file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/config-konsole.h"
"#pragma once
#define KONSOLE_VERSION \"plugin-build\"
#define HAVE_OS_DRAGONFLYBSD 0
#define WITH_X11 1
#define HAVE_DBUS 1
#define ENABLE_DECRQCRA 0
#define HAVE_GETPWUID 1
#define HAVE_MALLOC_TRIM 1
#define HAVE_KAPSULE 0
")

file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/konsoleprivate_export.h"
"#pragma once
#define KONSOLEPRIVATE_EXPORT __attribute__((visibility(\"default\")))
#define KONSOLEPRIVATE_NO_EXPORT __attribute__((visibility(\"hidden\")))
#define KONSOLEPRIVATE_DEPRECATED __attribute__((deprecated))
#define KONSOLEPRIVATE_DEPRECATED_EXPORT KONSOLEPRIVATE_EXPORT KONSOLEPRIVATE_DEPRECATED
#define KONSOLEPRIVATE_DEPRECATED_NO_EXPORT KONSOLEPRIVATE_NO_EXPORT KONSOLEPRIVATE_DEPRECATED
")

add_library(konsole_pasteimageplugin MODULE PasteImagePlugin.cpp)

# Konsole's headers use "../foo/bar.h" relative includes assuming the
# translation unit lives inside a src/ subdirectory. Mirror that by adding
# every src/* subdirectory as an include path.
file(GLOB _konsole_src_subdirs LIST_DIRECTORIES true "${KONSOLE_SOURCE_DIR}/src/*")
set(KONSOLE_INCLUDE_DIRS "${KONSOLE_SOURCE_DIR}/src")
foreach(_d IN LISTS _konsole_src_subdirs)
    if(IS_DIRECTORY "${_d}")
        list(APPEND KONSOLE_INCLUDE_DIRS "${_d}")
    endif()
endforeach()

target_include_directories(konsole_pasteimageplugin PRIVATE
    "${CMAKE_CURRENT_BINARY_DIR}"
    ${KONSOLE_INCLUDE_DIRS}
)

# Arch ships only versioned sonames (e.g. libkonsoleapp.so.26.04.1), no
# unversioned dev symlink, so glob and pick the highest.
file(GLOB KONSOLEAPP_CANDIDATES     "/usr/lib/libkonsoleapp.so.*")
file(GLOB KONSOLEPRIVATE_CANDIDATES "/usr/lib/libkonsoleprivate.so.*")
list(SORT KONSOLEAPP_CANDIDATES)
list(SORT KONSOLEPRIVATE_CANDIDATES)
list(GET KONSOLEAPP_CANDIDATES     -1 KONSOLEAPP_LIB)
list(GET KONSOLEPRIVATE_CANDIDATES -1 KONSOLEPRIVATE_LIB)
if(NOT KONSOLEAPP_LIB OR NOT KONSOLEPRIVATE_LIB)
    message(FATAL_ERROR "Could not find libkonsoleapp.so.* or libkonsoleprivate.so.* in /usr/lib")
endif()
message(STATUS "Linking against ${KONSOLEAPP_LIB}")
message(STATUS "Linking against ${KONSOLEPRIVATE_LIB}")

target_link_libraries(konsole_pasteimageplugin PRIVATE
    Qt6::Core Qt6::Gui Qt6::Widgets
    KF6::CoreAddons KF6::I18n KF6::XmlGui
    "${KONSOLEAPP_LIB}" "${KONSOLEPRIVATE_LIB}"
)

install(TARGETS konsole_pasteimageplugin
    DESTINATION "${KDE_INSTALL_PLUGINDIR}/konsoleplugins")
